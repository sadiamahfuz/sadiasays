import { Link, graphql } from "gatsby"
import Excerpt from "../components/Excerpt"

<ul style={{listStyleType: 'none'}}>
  {props.data.allMdx.nodes.map(node => (
    <li key={node.id}>
      <Excerpt data={node} />
    </li>
  ))}
</ul>

export const pageQuery = graphql`
  query {
  allMdx(sort: {fields: frontmatter___date, order: DESC}, filter: {frontmatter: {category: {eq: "tech"}}}) {
    nodes {
      excerpt
      frontmatter {
        date(formatString: "DD MMM YYYY")
        title
        tags
        category
        rating
        thumbnail {
          relativePath
        }
      }
      id
      fields {
        slug
      }
    }
  }
}
`
